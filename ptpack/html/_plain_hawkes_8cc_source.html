634 void PlainHawkes::Simulate(const unsigned& n, const unsigned& num_sequences, std::vector<Sequence>& sequences)
635 {
636  Eigen::Map<Eigen::VectorXd> Lambda0_ = Eigen::Map<Eigen::VectorXd>(parameters_.segment(0, num_dims_).data(), num_dims_);
637 
638  Eigen::Map<Eigen::MatrixXd> Alpha_ = Eigen::Map<Eigen::MatrixXd>(parameters_.segment(num_dims_, num_dims_ * num_dims_).data(), num_dims_, num_dims_);
639 
640  Eigen::MatrixXd expsum = Eigen::MatrixXd::Zero(num_dims_, num_dims_);
641 
642  Eigen::VectorXd last_event_per_dim = Eigen::VectorXd::Zero(num_dims_);
643 
644  sequences = std::vector<Sequence>();
645 
646  for(unsigned sequenceID = 0; sequenceID < num_sequences; ++ sequenceID)
647  {
648  Sequence seq;
649 
650  unsigned eventID = 0;
651 
652  double t = RNG_.GetExponential(1.0 / Lambda0_.array().sum());
653 
654  Event event;
655  event.EventID = eventID;
656  event.SequenceID = sequenceID;
657  event.DimentionID = AssignDim(Lambda0_);
658  event.time = t;
659  event.marker = 0;
660  seq.Add(event);
661 
662  last_event_per_dim(event.DimentionID) = t;
663 
664  ++ eventID;
665 
666  unsigned last_event_dim = event.DimentionID;
667 
668  while(eventID <= n)
669  {
670  Eigen::MatrixXd temp = expsum;
671 
672  double lambda_star = Lambda0_.array().sum();
673 
674  UpdateExpSum(t, last_event_per_dim, temp);
675 
676  lambda_star += (Alpha_.array() * temp.array()).sum();
677 
678  while(true)
679  {
680  t += RNG_.GetExponential(1.0 / lambda_star);
681 
682  Eigen::MatrixXd temp = expsum;
683 
684  UpdateExpSum(t, last_event_per_dim, temp);
685 
686  Eigen::VectorXd intensity_dim = Lambda0_.array() + (Alpha_.array() * temp.array()).colwise().sum().transpose();
687 
688  double D = RNG_.GetUniform();
689 
690  if(D <= intensity_dim.array().sum() / lambda_star)
691  {
692  Event event;
693  event.EventID = eventID;
694  event.SequenceID = sequenceID;
695  event.DimentionID = AssignDim(intensity_dim);
696  last_event_dim = event.DimentionID;
697 
698  if(last_event_per_dim(last_event_dim) > 0)
699  {
700  expsum.row(last_event_dim) = (-(t - last_event_per_dim(last_event_dim)) * Beta_.row(last_event_dim).array()).exp() * (1 + expsum.row(last_event_dim).array());
701  }
702 
703  event.time = t;
704  last_event_per_dim(event.DimentionID) = t;
705 
706  event.marker = 0;
707  seq.Add(event);
708 
709  ++ eventID;
710 
711  break;
712 
713  }else
714  {
715  lambda_star = intensity_dim.array().sum();
716  }
717  }
718  }
