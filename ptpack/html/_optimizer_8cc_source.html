194 void Optimizer::lbfgs(const Eigen::VectorXd& g, const Eigen::MatrixXd& s, const Eigen::MatrixXd& y, const double& Hdiag, Eigen::VectorXd& d)
195 {
196  unsigned p = s.rows(), k = s.cols();
197 
198  Eigen::VectorXd ro(k);
199 
200  for(unsigned i = 0; i < k; ++ i)
201  {
202  ro(i) = 1 / (y.col(i).transpose() * s.col(i));
203  }
204 
205  Eigen::MatrixXd q = Eigen::MatrixXd::Zero(p, k + 1);
206  Eigen::MatrixXd r = Eigen::MatrixXd::Zero(p, k + 1);
207  Eigen::VectorXd al = Eigen::VectorXd::Zero(k);
208  Eigen::VectorXd be = Eigen::VectorXd::Zero(k);
209 
210  q.col(k) = g;
211 
212  for(int i = k - 1; i >= 0; --i)
213  {
214  al(i) = ro(i) * s.col(i).transpose() * q.col(i + 1);
215  q.col(i) = q.col(i + 1) - al(i) * y.col(i);
216  }
217 
218  // Multiply by Initial Hessian
219  r.col(0) = Hdiag * q.col(0);
220 
221  for(unsigned i = 0; i < k; ++ i)
222  {
223  be(i) = ro(i) * y.col(i).transpose() * r.col(i);
224  r.col(i + 1) = r.col(i) + s.col(i) * (al(i) - be(i));
225  }
226 
227  d = r.col(k);
228 }
