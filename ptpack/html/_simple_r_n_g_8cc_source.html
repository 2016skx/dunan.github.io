91 double SimpleRNG::GetGamma(double shape, double scale)
92 {
93  // Implementation based on "A Simple Method for Generating Gamma Variables"
94  // by George Marsaglia and Wai Wan Tsang. ACM Transactions on Mathematical Software
95  // Vol 26, No 3, September 2000, pages 363-372.
96 
97  double d, c, x, xsquared, v, u;
98 
99  if (shape >= 1.0)
100  {
101  d = shape - 1.0/3.0;
102  c = 1.0/sqrt(9.0*d);
103  for (;;)
104  {
105  do
106  {
107  x = GetNormal();
108  v = 1.0 + c*x;
109  }
110  while (v <= 0.0);
111  v = v*v*v;
112  u = GetUniform();
113  xsquared = x*x;
114  if (u < 1.0 -.0331*xsquared*xsquared || log(u) < 0.5*xsquared + d*(1.0 - v + log(v)))
115  return scale*d*v;
116  }
117  }
118  else if (shape <= 0.0)
119  {
120  std::stringstream os;
121  os << "Shape parameter must be positive." << "\n"
122  << "Received shape parameter " << shape;
123  throw std::invalid_argument( os.str() );
124  }
125  else
126  {
127  double g = GetGamma(shape+1.0, 1.0);
128  double w = GetUniform();
129  return scale*g*pow(w, 1.0/shape);
130  }
131 }

244 int SimpleRNG::PoissonLarge(double lambda)
245 {
246  // "Rejection method PA" from "The Computer Generation of Poisson Random Variables" by A. C. Atkinson
247  // Journal of the Royal Statistical Society Series C (Applied Statistics) Vol. 28, No. 1. (1979)
248  // The article is on pages 29-35. The algorithm given here is on page 32.
249 
250  double c = 0.767 - 3.36/lambda;
251  double beta = PI/sqrt(3.0*lambda);
252  double alpha = beta*lambda;
253  double k = log(c) - lambda - log(beta);
254 
255  for(;;)
256  {
257  double u = GetUniform();
258  double x = (alpha - log((1.0 - u)/u))/beta;
259  int n = (int) floor(x + 0.5);
260  if (n < 0)
261  continue;
262  double v = GetUniform();
263  double y = alpha - beta*x;
264  double temp = 1.0 + exp(y);
265  double lhs = y + log(v/(temp*temp));
266  double rhs = k + n*log(lambda) - LogFactorial(n);
267  if (lhs <= rhs)
268  return n;
269  }
